import { Component, OnInit, Input } from '@angular/core';
import { faTimes, faCircleNotch } from '@fortawesome/free-solid-svg-icons';
import { Player } from '../player/player.model';

@Component({
  selector: 'app-cell',
  templateUrl: './cell.component.html',
  styleUrls: ['./cell.component.css']
})

/**
 * CellComponent class
 */
export class CellComponent {
  /**
   * Player that selected the cell
   */
  @Input() state: Player;

  /**
   * faTimes icon
   */
  faTimes = faTimes;

  /**
   * faCircleNotch icon
   */
  faCircleNotch = faCircleNotch;

  constructor() {
    this.state = null;
  }

}

<button type="button" class="btn btn-outline"
  [ngClass]="{'btn-outline-primary': state && state.colorClass === 'primary', 'btn-outline-secondary': state && state.colorClass === 'secondary'}">
  <span *ngIf="state && state.sign === 'faCircle'"><fa-icon [icon]="faCircleNotch" class="player-option"></fa-icon></span>
  <span *ngIf="state && state.sign === 'faTimes'"><fa-icon [icon]="faTimes" class="player-option"></fa-icon></span>
</button>
